Restrictions of the Set of Traces


restriction Eq_testing:
  "∀ x y #i. (Eq( x, y ) @ #i) ⇒ (x = y)"
  // safety formula

  /*
  expanded formula:
  "∀ x y #i. (Eq( x, y ) @ #i) ⇒ (x = y)"
  */

restriction InEq_testing:
  "∀ x y #i. (InEq( x, y ) @ #i) ⇒ (¬(x = y))"
  // safety formula

  /*
  expanded formula:
  "∀ x y #i. (InEq( x, y ) @ #i) ⇒ (¬(x = y))"
  */

restriction pairings_unique:
  "∀ id1 id2 role ki kr #i #j.
         ((Paired( id1, role, ki, kr ) @ #i) ∧
          (Paired( id2, role, ki, kr ) @ #j)) ⇒
         (#i = #j)"
  // safety formula

  expanded formula:
