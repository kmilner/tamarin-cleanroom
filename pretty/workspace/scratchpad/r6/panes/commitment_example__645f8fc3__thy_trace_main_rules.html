Fact Symbols with Injective Instances


St_I(id,=,=,.,.,.), St_R(id,=,=,.,.)


Multiset Rewriting Rules


rule (modulo AC) isend:
   [ !KU( x ) ] --[ K( x ) ]-> [ In( x ) ]

  /* has exactly the trivial AC variant */


rule (modulo AC) irecv:
   [ Out( x ) ] --> [ !KD( x ) ]

  /* has exactly the trivial AC variant */


rule (modulo E) RegisterKey:
   [ Fr( ~ltkA ) ]
  --[ Key( ~ltkA ) ]->
   [ !F_AgentKey( ~ltkA ), Out( pk(~ltkA) ) ]

  /* has exactly the trivial AC variant */
rule (modulo E) CompromiseKey:
   [ !F_AgentKey( ~ltk ) ]
  --[ Compromise( pk(~ltk) ) ]->
   [ Out( ~ltk ) ]

  /* has exactly the trivial AC variant */
rule (modulo E) State_Compromise:
   [ !F_TokenState( pkA, pkB, curtok ) ]
  --[ CompromiseToken( <pkA, pkB>, curtok ) ]->
   [ Out( curtok ) ]

  /* has exactly the trivial AC variant */
rule (modulo E) BindState_Init:
   [ !F_AgentKey( ~ltkA ), !F_AgentKey( ~ltkB ), Fr( ~id ) ]
  --[
  Paired( ~id, 'I', ~ltkA, ~ltkB ),
  Counter( ~id, '0' ),
  InvariantSource_I( ~id, ~ltkA, pk(~ltkB) )
  ]->
   [
   St_I( ~id, ~ltkA, pk(~ltkB), 'm1', 'uninitialized',
         sign('uninitialized', 'notoken')
   )
   ]

  /* has exactly the trivial AC variant */
rule (modulo E) BindState_Resp:
