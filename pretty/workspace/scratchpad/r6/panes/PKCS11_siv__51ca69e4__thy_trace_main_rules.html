Restrictions of the Set of Traces


restriction UniqueInteger:
  "∀ n #i #j.
         ((IsInteger( n ) @ #i) ∧ (IsInteger( n ) @ #j)) ⇒ (#i = #j)"
  // safety formula

  /*
  expanded formula:
  "∀ n #i #j.
         ((IsInteger( n ) @ #i) ∧ (IsInteger( n ) @ #j)) ⇒ (#i = #j)"
  */

restriction Lesser:
  "∀ x y #i. (LessThan( x, y ) @ #i) ⇒ (∃ z. (x++z) = y)"

  /*
  expanded formula:
  "∀ x y #i. (LessThan( x, y ) @ #i) ⇒ (∃ z. (x++z) = y)"
  */

restriction TrueIsTrue:
  "∀ x #i. (IsTrue( x ) @ #i) ⇒ (x = true)"
  // safety formula

  /*
  expanded formula:
  "∀ x #i. (IsTrue( x ) @ #i) ⇒ (x = true)"
  */
