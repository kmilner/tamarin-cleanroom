   [ !KU( x ) ] --[ K( x ) ]-> [ In( x ) ]

Restrictions of the Set of Traces


restriction Eq_testing:
  "∀ x y #i. (Eq( x, y ) @ #i) ⇒ (x = y)"
  // safety formula

  /*
  expanded formula:
  "∀ x y #i. (Eq( x, y ) @ #i) ⇒ (x = y)"
  */

restriction InEq_testing:
  "∀ x y #i. (InEq( x, y ) @ #i) ⇒ (¬(x = y))"
  // safety formula

  /*
  expanded formula:
  "∀ x y #i. (InEq( x, y ) @ #i) ⇒ (¬(x = y))"
  */

restriction pairings_unique:
  "∀ id1 id2 role ka kb #i #j.
         ((Paired( id1, role, ka, kb ) @ #i) ∧
          (Paired( id2, role, ka, kb ) @ #j)) ⇒
         (#i = #j)"
  // safety formula

  /*
  expanded formula:
  "∀ id1 id2 role ka kb #i #j.
         ((Paired( id1, role, ka, kb ) @ #i) ∧
          (Paired( id2, role, ka, kb ) @ #j)) ⇒
         (#i = #j)"
  */
